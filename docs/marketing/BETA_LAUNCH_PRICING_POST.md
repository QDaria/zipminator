## Countdown Widget Code

### JavaScript Live Countdown:

```javascript
// Zipminator Beta Countdown Widget
class ZipminatorCountdown {
  constructor() {
    this.quotas = {
      professional: { total: 1000, used: 0 },
      teams: { total: 500, used: 0 },
      enterprise: { total: 100, used: 0 }
    };

    // Update every 5 minutes from API
    this.updateInterval = 5 * 60 * 1000;
    this.init();
  }

  async init() {
    await this.fetchQuotas();
    this.renderCountdowns();
    setInterval(() => this.fetchQuotas(), this.updateInterval);
  }

  async fetchQuotas() {
    try {
      const response = await fetch('https://api.zipminator.zip/v1/beta/quotas');
      const data = await response.json();
      this.quotas = data;
      this.renderCountdowns();
    } catch (error) {
      console.error('Failed to fetch quotas:', error);
    }
  }

  renderCountdowns() {
    for (const [tier, quota] of Object.entries(this.quotas)) {
      const remaining = quota.total - quota.used;
      const percentage = (remaining / quota.total) * 100;

      const element = document.getElementById(`countdown-${tier}`);
      if (!element) continue;

      // Update text
      element.querySelector('.remaining').textContent = `${remaining}/${quota.total} spots remaining`;

      // Update color (green → yellow → red)
      let color = 'green';
      if (percentage < 50) color = 'yellow';
      if (percentage < 20) color = 'red';
      element.className = `countdown ${color}`;

      // Urgent badge if < 10% left
      if (percentage < 10) {
        element.classList.add('urgent');
        element.querySelector('.badge').textContent = '🔥 ALMOST GONE';
      }
    }
  }
}

// Initialize on page load
document.addEventListener('DOMContentLoaded', () => {
  new ZipminatorCountdown();
});
```
